A行の変数とB行の変数の共分散を変数の平均を計算することで求めるEXCELのマクロのコード例

    
Dim k As Integer
Dim n As Integer
Dim a As Double
Dim b As Double
Dim c As Double
Dim d As Double
a = 0
b = 0
c = 0
k = 1
While Cells(1,k).value <> ""
a = a + Cells(1,k)
b = b + Cells(2,k)
c = c + Cells(1,k) * Cells(2,k)
n = k
k = k + 1
Wend
d = c/n - a/n * b/n
MsgBox d
Cells(3,k) = d
